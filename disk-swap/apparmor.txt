#include <tunables/global>

profile disk_swap flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability sys_rawio,
  capability sys_admin,
  capability dac_override,
  capability dac_read_search,
  # DinD sandbox: layer extraction (chown), bridge networking, container user switching,
  # oom_score_adj management (sys_resource needed because Supervisor sets adj=200 on add-ons)
  capability chown,
  capability sys_chroot,
  capability sys_resource,
  capability net_admin,
  capability setuid,
  capability setgid,
  file,
  # Allow inner containerd/runc to escape to unconfined for container namespace setup.
  # Required for DinD: without this, setns fails when inner runc tries to enter a
  # container's network namespace (exit status 1 from containerd-shim-runc-v2).
  change_profile -> unconfined,
  # Allow ptrace-style operations needed by containerd-shim for namespace management.
  ptrace (read, trace),
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Add-on data
  /data/** rwix,

  # Static web UI files
  /var/www/** r,

  # Block device access â€” required for flashing USB devices
  /dev/sd* rw,
  /dev/sd** rw,
  /dev/mmcblk* rw,
  /dev/mmcblk** rw,
  /dev/nvme* r,
  /dev/nvme** r,
  /sys/block/** r,
  /sys/class/block/** r,
  /sys/devices/** r,
  /proc/partitions r,

  # Block device tools
  /usr/bin/lsblk ix,
  /usr/bin/blkid ix,
  /usr/sbin/blockdev ix,
  /usr/bin/dd ix,
  /bin/dd ix,
  /usr/bin/pv ix,

  # Download tools
  /usr/bin/curl ix,
  /usr/bin/wget ix,

  # Decompression
  /usr/bin/xz ix,
  /usr/bin/gzip ix,
  /usr/bin/gunzip ix,

  # Network access (for downloading images)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # Bun runtime
  /usr/local/bin/bun ix,
  /usr/src/** r,

  # Partition management (stages 3-4)
  /usr/sbin/partprobe ix,
  /usr/bin/mount ix,
  /usr/bin/umount ix,
  mount,
  umount,
  pivot_root,
  /usr/sbin/e2fsck ix,
  /usr/bin/findmnt ix,

  # Backup volume access
  /backup/** rw,

  # Loop device for partition mounting (avoids page cache aliasing)
  /dev/loop* rw,
  /dev/loop-control rw,

  # Temp mount point for target device
  /mnt/newsd/** rw,
}
